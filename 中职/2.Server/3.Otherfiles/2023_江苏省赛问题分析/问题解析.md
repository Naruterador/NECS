#### 解密
```shell
openssl enc -base64 -des3 -d -pbkdf2  -salt -iter 5 -in skills.enc -out skills.dec

## 使用私钥签名:
$ openssl rsautl -sign -inkey rsa_private_key2.pem -in skills.enc  -out skills.sig
## 使用公钥验证:
$ openssl rsautl -verify -inkey rsa_public_key2.pem -pubin -in skills.sig -out skills2.dec



```

#### DNS ACL
```shell
#named.conf配置
acl "js"
{
        192.168.107.101;

};


acl "bj"
{
        192.168.107.54;

};


acl "sh"
{
        192.168.107.55;

};

include "/etc/named.rfc1912.zones.js";
include "/etc/named.rfc1912.zones.bj";
include "/etc/named.rfc1912.zones.sh";

view default
{
match-clients { any; };
zone "." IN {
        type hint;
        file "named.ca";
  };

zone "skills.com" IN {
        type master;
        file "named.skills";
        allow-update { none; };
  };

};

#区域数据文件配置其中之一配置
[root@localhost opt]# cat /etc/named.rfc1912.zones.js
view js
{
match-clients { js; };
zone "skills.com" IN {
        type master;
        file "named.skills.js";
        allow-update { none; };
  };
};
```
